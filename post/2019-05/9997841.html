import java.util.Arrays;

public class MissingLetters
{
    public String getMissingLetters(String sentence) {
        sentence = sentence.toUpperCase();

        String res = "";
        for (char c = 'A'; c <= 'Z'; c++)
            if (sentence.indexOf(c) == -1)
                res += c;

        return res;
    }
}

4.1-2
Write pseudocode for the brute-force method of solving the maximum-subarray
problem. Your procedure should run in Θ(n²) time.


BRUTE-FORCE(A, low, high)
bf-sum = -∞
for i=low to high
   sum = 0
   for j=i to high
       sum = sum + A[j]
       if sum > bf-sum
          bf-sum = sum
          max-left = i
          max-right = j
return max-left, max-right, bf-sum

public static void main(String[] args)

static MaximumSubarray findMS(int[] A, int low, int high)
{
int mid = 0;

if (low == high)
{
return new MaximumSubarray(low, high, A[low]);
}
else
{
mid = (low+high)/2;

MaximumSubarray left = findMS(A, low, mid);

MaximumSubarray right = findMS(A, mid+1, high);

MaximumSubarray cross = findCS(A, low, mid, high);

if (left.sum >= right.sum && left.sum >= cross.sum)
{
return new MaximumSubarray(left.low, left.high, left.sum);
}
else if (right.sum >= left.sum && right.sum >= cross.sum)
{
return new MaximumSubarray(right.low, right.high, right.sum);
}
else
{
return new MaximumSubarray(cross.low, cross.high, cross.sum);
}
}
}

static MaximumSubarray findCS(int[] A, int low, int mid, int high)
{
int sum=0, left_sum=Integer.MIN_VALUE, right_sum=Integer.MIN_VALUE;
int max_left=0, max_right=0;

for (int i = mid; i >= low; i--)
{
sum += A[i];

if (sum > left_sum)
{
left_sum = sum;
max_left = i;
}
}

sum = 0;

for (int j = mid+1; j <= high; j++)
{
sum += A[j];

if (sum > right_sum)
{
right_sum = sum;
max_right = j;
}
}

return new MaximumSubarray(max_left, max_right, left_sum + right_sum);
}

static MaximumSubarray bruteforce(int[] A, int low, int high)
{
int left=0, right=0, sum=0, max_sum=Integer.MIN_VALUE;

for (int i = low; i <= high; i++)
{
sum = 0;

for (int j = i; j <= high ; j++)
{
sum += A[j];

if (sum > max_sum)
{
max_sum = sum;
left = i;
right = j;
}
}
}

return new MaximumSubarray(left, right, max_sum);
}

static MaximumSubarray mixed_br(int[] A, int low, int high)
{
int mid = 0;

if (high - low < crossover_point)
{
return bruteforce(A, low, high);
}
else
{
mid = (low+high)/2;

MaximumSubarray left = findMS(A, low, mid);

MaximumSubarray right = findMS(A, mid+1, high);

MaximumSubarray cross = findCS(A, low, mid, high);

if (left.sum >= right.sum && left.sum >= cross.sum)
{
return new MaximumSubarray(left.low, left.high, left.sum);
}
else if (right.sum >= left.sum && right.sum >= cross.sum)
{
return new MaximumSubarray(right.low, right.high, right.sum);
}
else
{
return new MaximumSubarray(cross.low, cross.high, cross.sum);
}
}
}
}

static MaximumSubarray findMS(int[] A, int low, int high)
{

if (low == high)
{
if (A[low] < 0)
{
return new MaximumSubarray(low, low, 0);
}
else
{
return new MaximumSubarray(low, high, A[low]);
}
}
else
{
int mid = (low+high)/2;

MaximumSubarray left = findMS(A, low, mid);

MaximumSubarray right = findMS(A, mid+1, high);

MaximumSubarray cross = findCS(A, low, mid, high);

if (left.sum >= right.sum && left.sum >= cross.sum)
{
return new MaximumSubarray(left.low, left.high, left.sum);
}
else if (right.sum >= left.sum && right.sum >= cross.sum)
{
return new MaximumSubarray(right.low, right.high, right.sum);
}
else
{
return new MaximumSubarray(cross.low, cross.high, cross.sum);
}
}
}

static MaximumSubarray findCS(int[] A, int low, int mid, int high)
{
int sum=0, left_sum=Integer.MIN_VALUE, right_sum=Integer.MIN_VALUE;
int max_left=0, max_right=0;

for (int i = mid; i >= low; i--)
{
sum += A[i];

if (sum > left_sum)
{
left_sum = sum;
max_left = i;
}
}

sum = 0;

for (int j = mid+1; j <= high; j++)
{
sum += A[j];

if (sum > right_sum)
{
right_sum = sum;
max_right = j;
}
}

if (left_sum + right_sum < 0)
{
return new MaximumSubarray(mid, mid, 0);
}
else
{
return new MaximumSubarray(max_left, max_right, left_sum + right_sum);
}
}
}


4.1-5
Use the following ideas to develop a nonrecursive, linear-time algorithm for the maximum-subarray problem. Start at the left end of the array, and progress toward the right, keeping track of the maximum subarray seen so far. Knowing a maximum subarray of A[1..j], extend the answer to find a maximum subarray ending at index j+1 by using the following observation: a maximum subarray of A[1..j+1] is either a maximum subarray of A[1..j] or a subarray A[i..j+1], for some 1 ≤ i ≤ j+1. Determine a maximum subarray of the form A[i..j+1] in constant time based on knowing a maximum subarray ending at index j .


class MaximumSubarray
{
int left=0, right=0, sum=0;

public MaximumSubarray() {};

public MaximumSubarray(int low, int high, int sum)
{
this.left = low;
this.right = high;
this.sum = sum;
}
}

class FindMaximumSubarray_i
{
public static void main(String[] args) 
{
int[] A = {13, -3, -25, 20, -3, -16, -23, 18, 20, -7, 12, -5, -22, 15, -4, 7};

long sTime = System.nanoTime();
MaximumSubarray maxsub = findMS(A, 0, A.length-1);
long eTime = System.nanoTime() - sTime;
System.out.println("left : " + maxsub.left + ", right : " + 
maxsub.right + ", max_sum : " + maxsub.sum + " Time : " + eTime);
}

static MaximumSubarray findMS(int[] A, int low, int high)
{
int i=0;

MaximumSubarray[] ith = new MaximumSubarray[A.length];

for (i=0; i < A.length; i++)
{
ith[i] = new MaximumSubarray();
}

ith[0].left = low;
ith[0].right = high;
ith[0].sum = A[low];

for (i = low+1; i <= high; i++)
{
if (ith[i-1].sum < 0)
{
ith[i].left = i;
ith[i].right = i+1;
ith[i].sum = A[i];
}
else
{
ith[i].left = ith[i-1].left;
ith[i].right = i+1;
ith[i].sum = ith[i-1].sum + A[i];
}
}

MaximumSubarray ms = ith[0];

for (i = low+1; i <= high; i++)
{
if (ms.sum < ith[i].sum)
{
ms = ith[i];
}
}

return ms;
}
